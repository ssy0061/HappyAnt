// store에 변수 초기값
const initialState = {
  // user 관련 state
  isLogin: 'false',
  username: null,
};

// reducer
export default function reducer(action, state = initialState) {
  // user 로그인 로그아웃 리듀서
  switch (action.type) {
    case 'LOGIN_USER':
      return { ...state, isLogin: true };
    case 'LOGOUT_USER':
      return { ...state, isLogin: false };
    default:
      return state;
  }
}
